% Versnummer: eigen superscript-routine, groter dan kanttekening-marker,
% in accent-kleur.
\newcommand{\vrsnr}[1]{%
  \mbox{\raisebox{0.5ex}{\fontsize{6.2}{7.4}\selectfont\textbf{\textcolor{accent}{#1}}}}\,}

% Counter — \vrsrefs stept hem, \verspair leest hem om bij verzen met

\newcounter{vrsrefcount}

% Kanttekening = voetnoot. Inhoud klein.
\newcommand{\kt}[1]{\footnote{#1}}

% Kanttekeningen binnen parallelle versboxen: de SV en modernisatie hebben
% eigen tellers en worden onderaan de pagina als links/rechts-blok gezet.
\newcounter{svkt}
\newcounter{modkt}
\newcommand{\ktmarkstyle}[1]{%
  \mbox{\raisebox{0.55ex}{\fnmarksize\textcolor{accent}{#1}}}}
\newcommand{\ktmarkinline}[1]{%
  \nobreak\hspace{0.16em}\ktmarkstyle{#1}\nobreak\hspace{0.16em}}
\newcommand{\ktmarksv}[1]{%
  \stepcounter{svkt}%
  \expandafter\xdef\csname ktfn@#1\endcsname{\arabic{svkt}}%
  \ktmarkinline{\arabic{svkt}}}
\newcommand{\ktmarkmod}[1]{%
  \stepcounter{modkt}%
  \expandafter\xdef\csname ktfn@#1\endcsname{\arabic{modkt}}%
  \ktmarkinline{\arabic{modkt}}}
\newcommand{\ktitem}[2]{%
  \noindent\hbox to 1.2em{\hss\ktmarkstyle{\csname ktfn@#1\endcsname}\hskip 0.25em}#2\par}
\newcommand{\ktnotecell}[1]{%
  \vtop{%
    \hsize=\parallelcolwidth
    \linewidth=\parallelcolwidth
    \parindent=0pt
    \parskip=1.4pt plus 0.4pt minus 0.2pt
    #1%
  }}
% Tussen opeenvolgende ktpairnotes-inserts plaatst TeX standaard géén
% separator-glue. Voeg aan het begin van elke insert een vskip toe gelijk
% aan parskip binnen ktnotecell, zodat inter-vers-gap == intra-vers-gap
% (geen lelijke sprongen tussen kanttekening-blokken van verschillende verzen).
\newcommand{\ktpairnotes}[2]{%
  \insert\footins{%
    \footnotesize
    \interlinepenalty\interfootnotelinepenalty
    \splittopskip\footnotesep
    \splitmaxdepth\dp\strutbox
    \floatingpenalty=20000
    \vskip 1.4pt plus 0.4pt minus 0.2pt
    \noindent\hbox to \textwidth{\ktnotecell{#1}\hfil\ktnotecell{#2}}%
    \par}}

\newcommand{\bref}[1]{{\textit{#1}}}

% Parallelle kolombreedte. De versrij zelf is ondeelbaar, zodat links en
% rechts nooit over verschillende pagina's uit elkaar lopen.
\newlength{\parallelcolwidth}
\setlength{\parallelcolwidth}{\dimexpr(\textwidth-\columnsep)/2\relax}

\newcommand{\parallelcell}[1]{%
  \vtop{%
    \hsize=\parallelcolwidth
    \linewidth=\parallelcolwidth
    \parindent=0pt
    \parskip=0.4pt plus 0.2pt minus 0.1pt
    #1\par
  }}

\newcommand{\parallelpair}[2]{%
  \par\noindent
  \hbox to \textwidth{\parallelcell{#1}\hfil\parallelcell{#2}}%
  \par}

% Intro/epilog blok over twee gesynchroniseerde kolommen.
\newcommand{\introblok}[2]{%
  \parallelpair{\footnotesize\textit{#1}}{\footnotesize\textit{#2}}%
  \par\vspace{1.2mm}}

\newcommand{\epilogblok}[2]{%
  \par\vspace{5mm}\introblok{#1}{#2}}

% Epilog onder hoofdstuk: gecentreerd, beide versies onder elkaar,
% los van de versrijen (om ‘gek-geplaatst’-effect te vermijden).
\newcommand{\epilogcenter}[2]{%
  \par\vspace{4mm}%
  \begin{center}%
    \rule{30mm}{0.3pt}\par\vspace{1.5mm}%
    {\small\textit{#1}}\par\vspace{1mm}%
    {\small\textit{#2}}%
  \end{center}%
  \par\vspace{2mm}}

% Kolomkop boven elke kolom binnen een hoofdstuk.
\newcommand{\kolomkop}[1]{%
  {\fontsize{7.2}{8.4}\selectfont\textsc{\textcolor{accent}{#1}}}\par\vspace{0.6mm}}

% Verspair: één ondeelbare horizontale rij. Page break valt tussen verzen;
% kanttekeningen worden onderaan de pagina verzameld.
% Voor verzen met kanttekeningen reserveert build_book.py vooraf genoeg
% verticale ruimte, zodat de noot niet naar de volgende pagina overloopt.
\newcommand{\ktneedspace}[1]{%
  \par
  \begingroup
    \dimen0=#1\baselineskip
    \dimen2=\pagegoal
    \advance\dimen2 by -\pagetotal
    \ifdim\dimen2<\dimen0
      \ifdim\pagetotal>0pt\pagebreak[4]\fi
    \fi
  \endgroup}

% Args: #1=nr #2=SVbody #3=SVrefs(may be empty) #4=Mbody #5=Mrefs(empty)
\newcommand{\verspair}[5]{%
  \setcounter{vrsrefcount}{0}%
  \parallelpair{\vrsnr{#1}#2#3}{\vrsnr{#1}#4#5}%
  \vspace{0.5pt}%
  \ifnum\value{vrsrefcount}>0\nopagebreak\vskip 1.6pt plus 0.3pt minus 0.2pt\fi}

% Header-pair boven verzen in een hoofdstuk (kolomkoppen).
\newcommand{\headerpair}[2]{%
  \parallelpair{\kolomkop{#1}}{\kolomkop{#2}}}
